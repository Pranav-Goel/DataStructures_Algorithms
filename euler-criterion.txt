long powmod(long x,long e,long mod)
    {
    long res;
    if(e==0)
        res=1;
    else if(e==1)
        res=x;
        else
        {
        res=powmod(x,e/2,mod);
        res=res*res%mod;
        if(e%2)
            res=res*x%mod;
    }
    return res;
    
}
